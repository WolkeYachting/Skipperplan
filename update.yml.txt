name: Skipperplan täglich aktualisieren

on:
  schedule:
    - cron: '0 1 * * *'   # Jeden Tag um 06:00 Uhr deutscher Zeit (05:00 UTC)
  workflow_dispatch:        # Auch manuell auslösbar über GitHub UI

jobs:
  update:
    runs-on: ubuntu-latest
    steps:
      - name: Repository auschecken
        uses: actions/checkout@v4

      - name: Excel vom Backend herunterladen
        run: |
          curl -f -o skipperplan.xlsx \
            "https://skipperplan-backend.onrender.com/download?password=${{ secrets.APP_PASSWORD }}"

      - name: Excel ins Repository committen
        run: |
          git config user.name  "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add skipperplan.xlsx
          git diff --staged --quiet || git commit -m "Skipperplan aktualisiert: $(date +'%d.%m.%Y')"
          git push
